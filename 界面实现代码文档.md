### 3.3 样式加载工具函数

创建 `utils.h` 中的样式加载函数：

```cpp
/**
 * @brief 从QSS文件加载全局样式表
 * @param filePath 样式文件路径
 * @return 样式表内容
 */
inline QString loadStyleFromFile(const QString &filePath)
{
    QFile file(filePath);
    if (!file.open(QFile::ReadOnly | QFile::Text)) {
        qWarning() << "无法打开样式文件：" << filePath;
        return QString();
    }

    QTextStream stream(&file);
    QString style = stream.readAll();
    file.close();

    return style;
}
```

### 3.4 更新后的 main.cpp

```cpp
#include "mainwindow.h"
#include "initfile.h"
#include "initfiledialog.h"
#include <QApplication>
#include <QFile>
#include <QDir>

int main(int argc, char *argv[])
{
    QApplication a(argc, argv);

    // 从文件加载全局样式
    QString stylePath = QApplication::applicationDirPath() + "/styles/global.qss";
    QString style = loadStyleFromFile(stylePath);
    if (!style.isEmpty()) {
        a.setStyleSheet(style);
    } else {
        qWarning() << "样式文件加载失败，使用默认样式";
    }

    // 初始化配置文件
    InitFile config;
    if (!config.loadConfig()) {
        qDebug() << QStringLiteral("配置文件加载失败，打开配置初始化窗口");

        // 显示配置初始化对话框
        InitFileDialog initDialog;
        if (initDialog.exec() != QDialog::Accepted) {
            return 0; // 用户取消配置，退出程序
        }

        // 重新加载配置
        config.loadConfig();
    }

    // 创建并显示登录窗口
    MainWindow w(&config);
    w.show();

    return QApplication::exec();
}
```

```cpp
void MainWindow::initDatabase()
{
    dbManager = new DatabaseManager();

    // 连接数据库状态信号
    connect(dbManager, &DatabaseManager::connectionStatusChanged,
            this, &MainWindow::on_DatabaseConnectionChanged);

    // 初始化数据库
    QString host = config->getDbHost();
    int port = config->getDbPort();
    QString dbname = config->getDbName();
    QString username = config->getDbUsername();
    QString password = config->getDbPassword();

    if (!dbManager->connectDatabase(host, port, dbname, username, password)) {
        QMessageBox::warning(this, QStringLiteral("警告"),
                           QStringLiteral("数据库连接失败，请检查配置信息！"));
    }

    updateConnectionStatus();

void MainWindow::on_loginButton_clicked()
{
    // 获取输入框的值
    QString username = ui->usernameEdit->text().trimmed();
    QString password = ui->passwordEdit->text();

    // 验证输入
    if (username.isEmpty() || password.isEmpty()) {
        QMessageBox::warning(this, QStringLiteral("警告"),
                           QStringLiteral("用户名和密码不能为空！"));
        return;
    }

    // 对密码进行SHA256加密
    QString encryptedPassword = encryptPassword(password);

    // 连接数据库验证
    QString userRole;
    if (dbManager->validateUser(username, encryptedPassword, userRole)) {
        QMessageBox::information(this, QStringLiteral("成功"),
                               QStringLiteral("登录成功！欢迎%1").arg(username));

        // 跳转到主功能界面
        showCarWindow();

        QMessageBox::warning(this, QStringLiteral("失败"),
                           QStringLiteral("用户名或密码错误！"));
    }
}
```

```cpp
void RegisterWindow::on_registerButton_clicked()
{
    if (!validateInput()) {
        return;
    }

    // 获取输入值
    QString username = ui->usernameEdit->text().trimmed();
    QString password = ui->passwordEdit->text();
    QString name = ui->nameEdit->text().trimmed();
    QString phone = ui->phoneEdit->text().trimmed();

    // 检查用户名是否已存在
    if (dbManager->isUsernameExists(username)) {
        showError(QStringLiteral("用户名已存在，请选择其他用户名！"));
        return;
    }

    // 对密码进行SHA256加密
    QString encryptedPassword = encryptPassword(password);

    // 注册用户
    if (dbManager->registerUser(username, encryptedPassword, phone)) {
        QMessageBox::information(this, QStringLiteral("成功"),
                               QStringLiteral("注册成功！"));

bool RegisterWindow::validateInput()
{
    QString username = ui->usernameEdit->text().trimmed();
    QString password = ui->passwordEdit->text();
    QString confirmPassword = ui->confirmPasswordEdit->text();
    QString name = ui->nameEdit->text().trimmed();
    QString phone = ui->phoneEdit->text().trimmed();

    // 检查用户名
    if (username.isEmpty()) {
        showError(QStringLiteral("用户名不能为空！"));
        ui->usernameEdit->setFocus();
        return false;
    }

    if (username.length() < 3 || username.length() > 20) {
        showError(QStringLiteral("用户名长度应为3-20个字符！"));
        ui->usernameEdit->setFocus();
        return false;
    }

    // 检查密码
    if (password.isEmpty()) {
        showError(QStringLiteral("密码不能为空！"));
        ui->passwordEdit->setFocus();
        return false;
    }

    if (password.length() < 6) {
        showError(QStringLiteral("密码长度不能少于6位！"));
        ui->passwordEdit->setFocus();
        return false;
    }

    // 检查确认密码
    if (password != confirmPassword) {
        showError(QStringLiteral("两次输入的密码不一致！"));
        ui->confirmPasswordEdit->setFocus();
        return false;
    }

    // 检查姓名
    if (name.isEmpty()) {
        showError(QStringLiteral("姓名不能为空！"));
        ui->nameEdit->setFocus();
        return false;
    }

    // 检查手机号
    if (phone.isEmpty()) {
        showError(QStringLiteral("手机号不能为空！"));
        ui->phoneEdit->setFocus();
        return false;
    }

    // 验证手机号格式
    QRegularExpression phoneRegex("^1[3-9]\\d{9}$");
    if (!phoneRegex.match(phone).hasMatch()) {
        showError(QStringLiteral("请输入正确的手机号！"));
        ui->phoneEdit->setFocus();
        return false;
    }

    return true;
}

void RegisterWindow::clearInput()

    ui->passwordEdit->clear();
    ui->confirmPasswordEdit->clear();
    ui->nameEdit->clear();
    ui->phoneEdit->clear();
}
```

```cpp
// ========== 槽函数实现 ==========

void CarWindow::onCheckInClicked()
{
    QString plate = input_checkin->text().trimmed();
    if (plate.isEmpty()) {
        QMessageBox::warning(this, QStringLiteral("警告"), QStringLiteral("请输入车牌号！"));
        return;
    }

    // 验证车牌号格式
    if (!Car::isValidLicensePlate(plate)) {
        QMessageBox::warning(this, QStringLiteral("警告"), QStringLiteral("车牌号格式不正确！"));
        return;
    }

    // 执行入库操作
    Car car;
    car.setLicensePlate(plate);
    car.setCheckInTime(QDateTime::currentDateTime());

    if (dbManager->insertCar(car)) {
        QMessageBox::information(this, QStringLiteral("成功"),
                               QStringLiteral("车辆 %1 入库成功！").arg(plate));
        input_checkin->clear();
        refreshCarTable();
        refreshStatistics();
    } else {
        QMessageBox::warning(this, QStringLiteral("失败"),
                           QStringLiteral("入库失败，请稍后重试！"));
    }
}

void CarWindow::onCheckOutClicked()
{
    QString plate = input_checkout->text().trimmed();
    if (plate.isEmpty()) {
        QMessageBox::warning(this, QStringLiteral("警告"), QStringLiteral("请输入车牌号！"));
        return;
    }

    // 执行出库操作
    Car car = dbManager->getCarByPlate(plate);
    if (car.getLicensePlate().isEmpty()) {
        QMessageBox::warning(this, QStringLiteral("警告"),
                           QStringLiteral("未找到该车辆的入库记录！"));
        return;
    }

    // 计算费用
    double hourlyRate = config->getParkFee();
    double fee = car.calculateFee(hourlyRate);

    // 更新出库信息
    car.setCheckOutTime(QDateTime::currentDateTime());
    car.setFee(fee);

    if (dbManager->updateCar(car)) {
        QMessageBox::information(this, QStringLiteral("成功"),
                               QStringLiteral("车辆 %1 出库成功！\n停车费用：¥%2")
                               .arg(plate).arg(fee, 0, 'f', 2));
        input_checkout->clear();
        refreshCarTable();
        refreshStatistics();
    } else {
        QMessageBox::warning(this, QStringLiteral("失败"),
                           QStringLiteral("出库失败，请稍后重试！"));
    }
}

void CarWindow::refreshStatistics()
{
    int total = config->getParkCount();
    int parked = dbManager->getParkedCarCount();
    int remaining = total - parked;
    double usage = total > 0 ? (double)parked / total * 100 : 0;

    label_parked->setText(QStringLiteral("在场车辆: %1/%2").arg(parked).arg(total));
    label_remaining->setText(QStringLiteral("剩余车位: %1/%2").arg(remaining).arg(total));
    label_usage->setText(QStringLiteral("使用率: %1%").arg(usage, 0, 'f', 1));
}

void CarWindow::refreshCarTable()

            table_cars->setItem(i, 3, new QTableWidgetItem("--"));
            table_cars->setItem(i, 4, new QTableWidgetItem("--"));
            table_cars->setItem(i, 5, new QTableWidgetItem(QStringLiteral("在场")));
        }
    }
}
```

```cpp
void InitFileDialog::onSubmitClicked()
{

        return;
    }

    // 保存配置

    config.setDbHost(ipEdit->text().trimmed());
    config.setDbPort(portEdit->text().toInt());
    config.setDbName(databaseEdit->text().trimmed());
    config.setDbUsername(accountEdit->text().trimmed());
    config.setDbPassword(passwordEdit->text());
    config.setParkingName(parkingNameEdit->text().trimmed());
    config.setParkFee(parkFeeEdit->text().toDouble());
    config.setParkCount(parkCountEdit->text().toInt());

    if (config.saveConfig()) {
        QMessageBox::information(this, QStringLiteral("成功"),
                               QStringLiteral("配置保存成功！"));
        accept();
    } else {
        showError(QStringLiteral("配置保存失败，请检查文件权限！"));
    }
```

```cpp
#ifndef UTILS_H
#define UTILS_H

#include <QString>
#include <QCryptographicHash>

/**
 * @brief 对密码进行SHA256加密
 * @param password 原始密码
 * @return 加密后的密码
 */
inline QString encryptPassword(const QString &password)
{
    QByteArray hash = QCryptographicHash::hash(
        password.toUtf8(),
        QCryptographicHash::Sha256
    );
    return hash.toHex();
}

/**
 * @brief 验证车牌号格式
 * @param plate 车牌号
 * @return 是否有效
 */
inline bool isValidLicensePlate(const QString &plate)
{
    // 简单验证：7-8位，第一位汉字，第二位字母
    if (plate.length() < 7 || plate.length() > 8) {
        return false;
    }

    QRegularExpression regex("^[京津沪渝冀豫云辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤川青藏琼宁][A-Z][A-Z0-9]{5,6}$");
    return regex.match(plate).hasMatch();
}

#endif // UTILS_H
```
